class Carte:
    def __init__(self,valeur,couleur):
        self.valeur=valeur
        self.couleur=couleur

    def jouer(self,other):
        if self.valeur==other.valeur:
            return "Egalité"
        elif self.valeur>other.valeur:
            return "Gagné"
        else:
            return "Perdu"
maCarte=Carte(8,"Coeur")
saCarte=Carte(6,"Trèfle")
jeu=maCarte.jouer(saCarte)
print(jeu)
