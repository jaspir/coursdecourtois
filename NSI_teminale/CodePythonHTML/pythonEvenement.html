import pygame
pygame.init() #lancement de pygame
width ,height=800,600  #Taille de la fenetre
monEcran=pygame.display.set_mode((width ,height )) #Affichage de la fenêtre

jeu_en_cours=True
n=0
down=False
while jeu_en_cours==True:
    pygame.display.update()
    monEcran.fill((100,40,70))

    #Un exemple de mouvement

    pygame.draw.line(monEcran,(0,0,0),(0,300),(n%800,300),10)
    n=n+1
    if down==True:
        pygame.draw.line(monEcran,(0,0,0),(300,0),(300,n%600),10)
    n=n+1
    for evenement in pygame.event.get():# Boucle sur les evenements
        if evenement.type==pygame.QUIT: # "appuyer sur le bouton quitter"
            pygame.quit()  #arret de pygame
            jeu_en_cours=False #arret de la boucle
        if evenement.type==pygame.KEYDOWN:# Appuis sur une fleche
            down=True
        else:
            down=False
